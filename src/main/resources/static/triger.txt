DELIMITER $$
DROP TRIGGER IF EXISTS trigger_update$$

CREATE TRIGGER trigger_update
before INSERT
ON usuarios FOR EACH ROW
BEGIN
     set NEW.update_at= now();
     set NEW.mod_user= now();
 END
 $$
 CREATE TRIGGER trigger_mod_user
before UPDATE
ON usuarios FOR EACH ROW
BEGIN

     set NEW.mod_user= now();
 END
 $$
