DELIMITER $$
DROP TRIGGER IF EXISTS trigger_update$$
CREATE TRIGGER trigger_update
before INSERT
ON usuarios FOR EACH ROW
BEGIN
     set NEW.update_at= now();
 END
 $$

 DELIMITER $$
DROP TRIGGER IF EXISTS trigger_update_cuentas$$
CREATE TRIGGER trigger_update_cuentas
before INSERT
ON cuentas FOR EACH ROW
BEGIN
     set NEW.update_at= CURRENT_TIMESTAMP();
 END
 $$
